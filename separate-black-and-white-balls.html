class Solution: def minimumSteps(self, s: str) -> int:

arr = list(map(int, s)) #print(arr) result = 0 j = len(arr) – 1 while(j >= 0 and arr[j] == 1): j -= 1 # off by 1 j += 1 temp = j-1 occ = 0 count = 0 while(temp >= 0): if arr[temp] == 1: result += (count – occ) occ += 1 temp -= 1 count += 1 return result
