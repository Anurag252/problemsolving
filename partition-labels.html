mp = {} for id, k in enumerate(s): mp[k] = id start = 0 end = 0 curr = 0 res = [] while(start <= end) : if curr len(s): return res end = max(end, mp[s[curr]]) if curr end: res.append(end – start + 1) start = end + 1 curr = end end += 1 curr += 1 return res
