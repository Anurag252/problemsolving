def generate_palindromes(): # Yield palindromes in increasing order for length in range(1, 100): # Adjust as needed half = 10 ** ((length – 1) // 2) for i in range(half, 10**((length + 1) // 2)): s = str(i) if length % 2 == 0: yield int(s + s[::-1]) else: yield int(s + s[-2::-1]) def to_base_k(n, k): digits = [] while n > 0: digits.append(str(n % k)) n //= k return ‘’.join(reversed(digits)) or ‘0’ found = 0 final = 0 total = 0 while(True): for t in generate_palindromes(): if to_base_k(t, k) to_base_k(t, k)[::-1]: total += t found += 1 if found n: return total
