class Solution: def mergeArrays(self, nums1: List[List[int]], nums2: List[List[int]]) -> List[List[int]]: mp = {}

mx = 0 for k in nums1: mp[k0] = k1 mx = max(mx, k0) for k in nums2: if k0 not in mp: mp[k0] = k1 else: mp[k0] += k1 mx = max(mx, k0) res = [] for k in range(1,mx+1): if k in mp: res.append([k, mp[k]]) return res
