#!/bin/sh #

  1. An example hook script to check the commit log message taken by
  2. applypatch from an e-mail message.
#
  1. The hook should exit with non-zero status after issuing an
  2. appropriate message if it wants to stop the commit. The hook is
  3. allowed to edit the commit message file.
#
  1. To enable this hook, rename this file to “applypatch-msg”.

. git-sh-setup
commitmsg=”$(git rev-parse —git-path hooks/commit-msg)“
test -x “$commitmsg” && exec “$commitmsg” ${1+”$@”} :
