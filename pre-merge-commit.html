#!/bin/sh #

  1. An example hook script to verify what is about to be committed.
  2. Called by “git merge” with no arguments. The hook should
  3. exit with non-zero status after issuing an appropriate message to
  4. stderr if it wants to stop the merge commit.
#
  1. To enable this hook, rename this file to “pre-merge-commit”.

. git-sh-setup
test -x “$GIT_DIR/hooks/pre-commit” && exec “$GIT_DIR/hooks/pre-commit“ :
