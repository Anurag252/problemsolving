class Solution: def countDays(self, days: int, meetings: List[List[int]]) -> int:

meetings.sort(key=lambda x : (x0, x1)) print(meetings) count = -1 start = 0 end = 0 for k in meetings: print(start, end, count) if end < k0: count += (end – start + 1) start = k0 end = k1 else: end = max(k1, end) print(start, end, count) count += (end – start + 1) return days – count
