1. Use the official Python image from the Docker Hub
    FROM python:3.9-slim
  1. Set environment variables
    ENV PYTHONDONTWRITEBYTECODE 1
    ENV PYTHONUNBUFFERED 1
  1. Set the working directory in the container
    WORKDIR /app
  1. Copy the dependencies file to the working directory
    COPY requirements.txt .
  1. Install any dependencies specified in requirements.txt
    RUN pip install -r requirements.txt
  1. Copy the Python script into the container at /app
    COPY convert_makrdown.py .
  1. Run the Python script
    CMD [“python”, “convert_makrdown.py”]
