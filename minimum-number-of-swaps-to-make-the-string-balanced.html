class Solution: def minSwaps(self, s: str) -> int: imbalance = 0 balance = 0

for char in s: if char '[': balance += 1 else: # char ‘]‘ balance -= 1 # If balance goes negative, we have more ] than [ so far if balance < 0: imbalance += 1 balance = 0 # Reset balance to zero to continue counting fresh # Each swap fixes two misplacements return (imbalance + 1) // 2
