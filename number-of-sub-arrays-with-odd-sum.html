pref_sum = [] temp = 0 for k in arr: temp += k pref_sum.append(temp) odds = [] evens = [] odd = 0 even = 0 for k in pref_sum: if k % 2 == 0: even += 1 else: odd += 1 # check zero odds.append(odd) evens.append(even) res = 0 for i, k in enumerate(pref_sum): if i 0: continue if k % 2 0: res += odds[i-1] else: res += evens[i-1] res = res % (pow(10,9)+ 7) return (res + odd) % (pow(10,9) + 7)
