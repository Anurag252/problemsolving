#!/bin/sh #

  1. An example hook script to verify what is about to be committed
  2. by applypatch from an e-mail message.
#
  1. The hook should exit with non-zero status after issuing an
  2. appropriate message if it wants to stop the commit.
#
  1. To enable this hook, rename this file to “pre-applypatch”.

. git-sh-setup
precommit=”$(git rev-parse —git-path hooks/pre-commit)“
test -x “$precommit” && exec “$precommit” ${1+”$@”} :
